\newcommand{\image}[3]{
\begin{figure}[!htb]
	\centering
	\includegraphics[width=#3\textwidth]{#1}
	\caption{#2}
\end{figure}
}

\newcommand{\centreq}[1]{
\begin{equation}
#1
\end{equation}
}

	\section{\centering#1}
}
\newcommand{\ocsection}[1]{
	\section*{\centering#1}
}
\newcommand{\csubsection}[1]{
	\subsection{\centering#1}
}
\newcommand{\ocsubsection}[1]{
	\subsection*{\centering#1}
}
